        <script>
            let a = document.querySelector('.circle');
            let x = 0;
            let y = 0;
            const UNIT = 10;

            a.onkeydown =function(event){
               

                
                    if(event.keyCode==37){ //왼쪽
                        if(x>=10){
                        x=x-UNIT;
                        a.style.left = x+"px";
                        }
                        
                    }else if(event.keyCode==38){//위
                        if(y>=10){
                        y=y-UNIT;
                        a.style.top = y+"px";
                        }
                        
                    }
                    else if(event.keyCode==39){//오른쪽
                        if(x<=290){
                        x=x+UNIT;
                        a.style.left = x+"px";
                        }
                    }
                    else if(event.keyCode==40){ //아래쪽 
                        if(y<=290){
                        y=y+UNIT;
                        a.style.top = y+"px";
                        }
                    }
                
            }//func
           
            
            
        </script>
    </body>
